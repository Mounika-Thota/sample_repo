import { test, expect } from '@playwright/test';

// Happy path smoke test - login and verify navigation menu
// The selectors used below are from real DOM exploration (session restored if needed)
test.describe('Core Navigation', () => {
  test('should login and display main navigation', async ({ page }) => {
    // Navigate to login page
    await page.goto('https://example-app-url.com/login', { waitUntil: 'domcontentloaded' });
    await expect(page).toHaveURL(/.*login/);
    
    // Fill login form (selectors discovered from snapshot DOM)
    await expect(page.locator('input[name="username"]')).toBeVisible();
    await page.locator('input[name="username"]').fill('admin');
    await expect(page.locator('input[name="password"]')).toBeVisible();
    await page.locator('input[name="password"]').fill('admin123');
    await page.locator('button:has-text("Login")').click();

    // Wait for main dashboard and navigation to appear
    await page.waitForLoadState('domcontentloaded');
    // Adaptation: Selector '.oxd-main-menu' and '.oxd-main-menu-item' are from a real OrangeHRM instance
    await expect(page.locator('.oxd-main-menu')).toBeVisible();
    // Validate at least one navigation link is present
    await expect(page.locator('.oxd-main-menu-item')).toHaveCountGreaterThan(0);
  });
});

// Example test for form interaction: Apply Leave
// Adaptation noted below

test.describe('Leave - Apply', () => {
  test('should open Apply Leave form and submit with valid data', async ({ page }) => {
    // Must be logged in - run login helper
    await page.goto('https://example-app-url.com/login', { waitUntil: 'domcontentloaded' });
    await page.locator('input[name="username"]').fill('employee');
    await page.locator('input[name="password"]').fill('employeePass!');
    await page.locator('button:has-text("Login")').click();
    await page.waitForLoadState('domcontentloaded');

    // Navigate to Leave -> Apply (verified in DOM: uses sidebar .oxd-main-menu-item)
    await page.locator('.oxd-main-menu-item:has-text("Leave")').click();
    await page.waitForTimeout(500); // Wait for submenu transition
    await page.locator('.oxd-main-menu-item:has-text("Apply")').click();
    await page.waitForLoadState('domcontentloaded');

    // Ensure Apply Leave form loads (label: Leave Type)
    await expect(page.locator('label:has-text("Leave Type")')).toBeVisible();

    // Adaptation: Real field class is '.oxd-select-text-input' (from DOM)
    // Open the leave type dropdown and select the first available option
    await page.locator('.oxd-select-text-input').click();
    await page.waitForSelector('.oxd-select-dropdown');
    const firstOption = page.locator('.oxd-select-dropdown .oxd-select-option').first();
    await firstOption.click();

    // Date pickers: Use '.oxd-date-input' for both from and to dates
    const dateInputs = page.locator('.oxd-date-input input');
    await expect(dateInputs.nth(0)).toBeVisible();
    await dateInputs.nth(0).fill('2024-07-01'); // Start date
    await expect(dateInputs.nth(1)).toBeVisible();
    await dateInputs.nth(1).fill('2024-07-01'); // End date

    // Comment field is <textarea>
    const commentBox = page.locator('textarea');
    if (await commentBox.count() > 0) {
      await commentBox.fill('Applying leave for medical reasons.');
    }

    // Adapt to actual button text: 'Apply'
    await page.locator('button:has-text("Apply")').click();

    // Wait for toast feedback (real selector: .oxd-toast)
    await page.waitForSelector('.oxd-toast');
    const toast = page.locator('.oxd-toast');
    await expect(toast).toBeVisible();
    await expect(toast.locator('.oxd-text--toast-title')).toHaveText(/Success|Applied/i);
  });
});

// Adapted negative test: Apply Leave with missing data
// Will use actual error message from .oxd-input-field-error-message

test.describe('Leave - Apply Negative', () => {
  test('should show error when required fields are missing', async ({ page }) => {
    await page.goto('https://example-app-url.com/login', { waitUntil: 'domcontentloaded' });
    await page.locator('input[name="username"]').fill('employee');
    await page.locator('input[name="password"]').fill('employeePass!');
    await page.locator('button:has-text("Login")').click();
    await page.waitForLoadState('domcontentloaded');
    await page.locator('.oxd-main-menu-item:has-text("Leave")').click();
    await page.waitForTimeout(500);
    await page.locator('.oxd-main-menu-item:has-text("Apply")').click();
    await page.waitForLoadState('domcontentloaded');

    // Attempt to submit form without filling fields
    await page.locator('button:has-text("Apply")').click();

    // Real-world error message selector
    const errorMsg = page.locator('.oxd-input-field-error-message');
    await expect(errorMsg).toBeVisible();
    // Assert on one of the default validation strings
    await expect(errorMsg).toHaveText(/Required|This field is required/i);
  });
});
