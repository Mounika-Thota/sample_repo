const baseUrl = 'https://opensource-demo.orangehrmlive.com/web/index.php/';

// Utility function: Login as Admin
async function loginAsAdmin(page: any, username: string = 'Admin', password: string = 'admin123') {
  await page.goto(baseUrl, { waitUntil: 'domcontentloaded' });
  // LOCATORS BELOW ARE PLACEHOLDERS. Replace using MCP-located refs only!
  await page.getByPlaceholder('Username').fill(username);
  await page.getByPlaceholder('Password').fill(password);
  await page.locator('button:has-text("Login")').click();
  await page.waitForURL('**/dashboard', { timeout: 10000 });
}
