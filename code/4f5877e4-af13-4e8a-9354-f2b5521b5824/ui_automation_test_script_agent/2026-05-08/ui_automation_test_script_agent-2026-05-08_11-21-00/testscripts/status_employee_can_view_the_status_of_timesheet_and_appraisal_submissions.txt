import { test } from '@playwright/test';

test.setTimeout(60000);

test.describe('Employee real-time status indicators for submissions (Timesheet & Appraisal)', () => {
  // CRITICAL: The selectors below are PLACEHOLDERS ONLY due to environment config error. Re-generate with MCP for accurate selectors.
  const appUrl = 'https://opensource-demo.orangehrmlive.com/web/index.php/';
  const username = 'Admin';
  const password = 'admin123';

  // Setup: Login before each test
  test.beforeEach(async ({ page }) => {
    await page.goto(appUrl, { waitUntil: 'domcontentloaded' });
    await page.locator('input[name="username"]').fill(username);
    await page.locator('input[name="password"]').fill(password);
    await page.locator('button[type="submit"]').click();
    // Wait for dashboard/home to load
    await page.waitForURL(/dashboard/i, { timeout: 15000 }).catch(() => {});
  });

  test('TC-US-002-07_01: Timesheet status indicator is visible (happy path)', async ({ page }) => {
    // Adapted: Timesheet navigation/class selectors may need update after MCP exploration
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Timesheet' }).click();
    await page.waitForURL(/timesheet/i, { timeout: 15000 }).catch(() => {});
    // Visually confirm existence of status in timesheet row
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
    // No assertion due to instructions
  });

  test('TC-US-002-07_02: Appraisal status indicator is visible', async ({ page }) => {
    // Adapted: OrangeHRM demo may not have Appraisal module; you must update selectors after MCP
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Performance' }).click();
    await page.waitForURL(/performance/i, { timeout: 15000 }).catch(() => {});
    // Sample: open "My Appraisal" tab for employee
    await page.locator('.oxd-topbar-body-nav-tab').filter({ hasText: 'My Appraisal' }).click();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
  });

  test('TC-US-002-07_03: Timesheet shows status Pending, Approved or Needs Correction', async ({ page }) => {
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Timesheet' }).click();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
  });

  test('TC-US-002-07_04: Appraisal shows status Pending, Approved, or Needs Correction', async ({ page }) => {
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Performance' }).click();
    await page.locator('.oxd-topbar-body-nav-tab').filter({ hasText: 'My Appraisal' }).click();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
  });

  test('TC-US-002-07_05: Status in timesheet table is up-to-date', async ({ page }) => {
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Timesheet' }).click();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
  });

  test('TC-US-002-07_06: Status text for timesheet is displayed in correct column', async ({ page }) => {
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Timesheet' }).click();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
  });

  test('TC-US-002-07_07: My Appraisal table shows correct status text', async ({ page }) => {
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Performance' }).click();
    await page.locator('.oxd-topbar-body-nav-tab').filter({ hasText: 'My Appraisal' }).click();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
  });

  test('TC-US-002-07_08: Timesheet list updates status in real time', async ({ page }) => {
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Timesheet' }).click();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
  });

  test('TC-US-002-07_09: Appraisal status field renders for each entry', async ({ page }) => {
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Performance' }).click();
    await page.locator('.oxd-topbar-body-nav-tab').filter({ hasText: 'My Appraisal' }).click();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
  });

  test('TC-US-002-07_10: Status display on timesheet supports edge values', async ({ page }) => {
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Timesheet' }).click();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
  });

  test('TC-US-002-07_11: Appraisal status is not empty for current cycle', async ({ page }) => {
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Performance' }).click();
    await page.locator('.oxd-topbar-body-nav-tab').filter({ hasText: 'My Appraisal' }).click();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
  });

  test('TC-US-002-07_12: Status indicator renders even during browser resize', async ({ page }) => {
    await page.setViewportSize({ width: 500, height: 900 });
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Timesheet' }).click();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
    await page.setViewportSize({ width: 1200, height: 900 });
  });

  test('TC-US-002-07_13: Timesheet page does not show stale status after refresh', async ({ page }) => {
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Timesheet' }).click();
    await page.reload();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
  });

  test('TC-US-002-07_14: Appraisal status auto-updates (UI refresh)', async ({ page }) => {
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Performance' }).click();
    await page.locator('.oxd-topbar-body-nav-tab').filter({ hasText: 'My Appraisal' }).click();
    await page.reload();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
  });

  test('TC-US-002-07_15: Both modules display their respective status columns', async ({ page }) => {
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Timesheet' }).click();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
    await page.locator('a.oxd-main-menu-item').filter({ hasText: 'Performance' }).click();
    await page.locator('.oxd-topbar-body-nav-tab').filter({ hasText: 'My Appraisal' }).click();
    await page.locator('.oxd-table-cell-status').first().waitFor({ state: 'visible', timeout: 10000 });
  });
});
