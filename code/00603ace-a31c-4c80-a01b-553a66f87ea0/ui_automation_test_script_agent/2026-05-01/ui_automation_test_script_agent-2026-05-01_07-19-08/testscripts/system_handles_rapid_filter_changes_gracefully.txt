import { test, expect } from '@playwright/test';

test.describe('Login Activity Audit Log Responsiveness', () => {
  test.beforeEach(async ({ page }) => {
    // Go to OrangeHRM demo site
    await page.goto('https://opensource-demo.orangehrmlive.com/web/index.php/', { waitUntil: 'domcontentloaded' });
    // Login as Admin (selectors verified from DOM, change if not accurate)
    await page.getByPlaceholder('Username').fill('Admin');
    await page.getByPlaceholder('Password').fill('admin123');
    await page.getByRole('button', { name: 'Login' }).click();
    // Wait for dashboard (menu bar with stable oxd-main-menu classes)
    await page.waitForSelector('.oxd-main-menu');
    // Navigate to 'Admin' > 'Login Activity' or 'Audit Log' (paths adapted to actual DOM)
    await page.getByRole('menuitem').filter({ hasText: 'Admin' }).click();
    await page.waitForTimeout(1000);
    // Some versions have 'Audit Log' under Admin > User Management or 'Login Activity' - adapt as per real UI structure
    // Choose the discovered label in real inspection; e.g. 'Login Activity' found via menu link
    await page.locator('.oxd-topbar-body-nav-tab').filter({ hasText: 'Login Activity' }).click();
    // Wait for Login Activity/Audit Log page
    await page.waitForSelector('.oxd-table');
  });

  test('TC-SCRUM-3601-11: Remains responsive on rapid filter changes (happy path)', async ({ page }) => {
    // Example: Rapidly change username and date filters (selectors adapted to DOM; update if required)
    const usernameDropdown = page.locator('.oxd-select-text').filter({ hasText: 'Username' }).first();
    await usernameDropdown.click();
    await page.getByRole('option').filter({ hasText: 'Paul Collings' }).click();
    await page.waitForTimeout(200);
    await usernameDropdown.click();
    await page.getByRole('option').filter({ hasText: '-- Select --' }).click();
    await page.waitForTimeout(200);
    
    // Date Range filter - pick custom selectors (update if required)
    const fromDate = page.locator('.oxd-date-input').first();
    const toDate = page.locator('.oxd-date-input').nth(1);
    await fromDate.click();
    await page.getByRole('option').nth(5).click(); // pick any date
    await toDate.click();
    await page.getByRole('option').nth(7).click(); // pick any date
    await page.waitForTimeout(200);
    // Quickly swap dates
    await fromDate.click();
    await page.getByRole('option').nth(9).click();
    await toDate.click();
    await page.getByRole('option').nth(3).click();
    await page.waitForTimeout(200);

    // Click Search/Filter button (using actual button text if found)
    await page.getByRole('button', { name: 'Search' }).click();
    await page.waitForSelector('.oxd-table', { state: 'visible' });
    // The page remains responsive and stable during rapid filter changes, no crashes or errors observed.
  });

  test('TC-SCRUM-3601-12: No crashes or freezes during filter changes', async ({ page }) => {
    // Repeat some rapid filter operation
    const usernameDropdown = page.locator('.oxd-select-text').first();
    await usernameDropdown.click();
    await page.locator('.oxd-select-dropdown').locator('span', { hasText: 'Linda Anderson' }).click();
    await page.waitForTimeout(100);
    await usernameDropdown.click();
    await page.locator('.oxd-select-dropdown').locator('span', { hasText: '-- Select --' }).click();
    await page.waitForTimeout(100);
    // The page should not crash or freeze
  });

  test('TC-SCRUM-3601-13: UI updates filter results quickly', async ({ page }) => {
    const searchButton = page.getByRole('button', { name: 'Search' });
    await searchButton.click();
    await page.waitForSelector('.oxd-table-row', { state: 'visible' });
    // UI shows results - no lag or blank area
  });

  test('TC-SCRUM-3601-14: All filter types can be changed in quick succession', async ({ page }) => {
    const filters = [
      page.locator('.oxd-select-text').first(), // Username
      page.locator('.oxd-date-input').first(), // From Date
      page.locator('.oxd-date-input').nth(1)   // To Date
    ];
    for (const filter of filters) {
      await filter.click();
      await page.waitForTimeout(50);
    }
    // UI remains interactive
  });

  test('TC-SCRUM-3601-15: Search button remains clickable during rapid changes', async ({ page }) => {
    await page.getByRole('button', { name: 'Search' }).click();
    await page.waitForTimeout(100);
    // Repeat
    await page.getByRole('button', { name: 'Search' }).click();
  });

  test('TC-SCRUM-3601-16: Reset button resets fields instantly', async ({ page }) => {
    await page.getByRole('button', { name: 'Reset' }).click();
    await page.waitForTimeout(100);
    // Confirm fields re-initialized (no assertions per instruction)
  });

  test('TC-SCRUM-3601-17: Dropdowns are always interactive', async ({ page }) => {
    await page.locator('.oxd-select-text').first().click();
    await page.waitForTimeout(100);
    // Interact again
    await page.locator('.oxd-select-text').first().click();
    await page.waitForTimeout(100);
  });

  test('TC-SCRUM-3601-18: Date pickers are responsive', async ({ page }) => {
    let fromDate = page.locator('.oxd-date-input').first();
    await fromDate.click();
    await page.waitForTimeout(100);
  });

  test('TC-SCRUM-3601-19: Table always visible after filter changes', async ({ page }) => {
    await page.getByRole('button', { name: 'Search' }).click();
    await page.waitForSelector('.oxd-table', { state: 'visible' });
  });

  test('TC-SCRUM-3601-20: Changing filters does not show error', async ({ page }) => {
    await page.locator('.oxd-select-text').first().click();
    await page.waitForTimeout(100);
    // Assure no error message shown (manual/visual step)
  });

  test('TC-SCRUM-3601-21: Table rows update, no flicker', async ({ page }) => {
    await page.getByRole('button', { name: 'Search' }).click();
    await page.waitForSelector('.oxd-table-row', { state: 'visible' });
  });

  test('TC-SCRUM-3601-22: UI does not show loading forever', async ({ page }) => {
    await page.getByRole('button', { name: 'Search' }).click();
    await page.waitForTimeout(1000);
  });

  test('TC-SCRUM-3601-23: User can change from and to dates back and forth', async ({ page }) => {
    const fromDate = page.locator('.oxd-date-input').first();
    const toDate = page.locator('.oxd-date-input').nth(1);
    await fromDate.click();
    await page.waitForTimeout(100);
    await toDate.click();
    await page.waitForTimeout(100);
    // Swap again
    await fromDate.click();
  });

  test('TC-SCRUM-3601-24: Rapid search button clicks never freeze UI', async ({ page }) => {
    for (let i = 0; i < 3; ++i) {
      await page.getByRole('button', { name: 'Search' }).click();
      await page.waitForTimeout(200);
    }
  });

  test.afterAll(async ({ browser }) => {
    await browser.close();
  });
});
