    // Wait for the activity/audit log table to be visible
    const dataTable = page.locator('.oxd-table, .oxd-table-card');
    await expect(dataTable).toBeVisible();

    // Locate the Username filter
    // In OrangeHRM 4.7+: filter row above table, input with placeholder or label "Username"
    const userInput = page.locator('input[placeholder="Username"], .oxd-input').nth(0);
    // Re-validate with field labels if multiple .oxd-input fields

    // Enter username to filter for: 'Admin' (demo value from test case)
    await expect(userInput).toBeVisible();
    await userInput.fill('Admin');

    // Find and click the filter/search/apply button (usually "Search", .oxd-button)
    // If pressing Enter does submit, use .press('Enter') as alternative
    const searchButton = page.locator('button').filter({ hasText: /Search|Apply|Filter/i });
    if (await searchButton.count()) {
      await searchButton.first().click();
    } else {
      await userInput.press('Enter');
    }

    // Wait for filtered results table to update
    await page.waitForTimeout(1000); // OrangeHRM sometimes animates tables; if debounce/async occurs
    // Validate that each row contains 'Admin' in Username column
    const resultRows = dataTable.locator('.oxd-table-row');
    const rowsCount = await resultRows.count();
    for (let i = 0; i < rowsCount; ++i) {
      const usernameCell = resultRows.nth(i).locator('div').nth(1); // OrangeHRM: usually username is second cell
      await expect(usernameCell).toContainText('Admin');
    }
    // -- If no results, validate the empty state message accurately.
  });
