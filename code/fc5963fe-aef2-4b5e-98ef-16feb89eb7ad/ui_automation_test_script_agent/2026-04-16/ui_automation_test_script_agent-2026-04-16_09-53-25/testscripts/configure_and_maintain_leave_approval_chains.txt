import { test } from '@playwright/test';

test.describe.configure({ mode: 'serial' });
test.setTimeout(60000);

const HR_URL = 'https://opensource-demo.orangehrmlive.com/';
const USERNAME = 'Admin';
const PASSWORD = 'admin123';

// Helper for login
async function login(page) {
  await page.goto(HR_URL);
  await page.getByRole('textbox', { name: 'Username' }).fill(USERNAME);
  await page.getByRole('textbox', { name: 'Password' }).fill(PASSWORD);
  await page.getByRole('button', { name: 'Login' }).click();
  await page.waitForURL('**/dashboard/index');
}

// Helper: navigate to Leave workflow config ("Leave Types" for this demo)
async function openLeaveWorkflowConfig(page) {
  await page.getByRole('link', { name: 'Leave' }).click();
  // Open Configure from Topbar
  await page.getByText('Configure').click();
  await page.getByRole('menuitem', { name: 'Leave Types' }).click();
  await page.waitForSelector('text=Leave Types');
}

// TC-US-001-01: HR User accesses leave workflow configuration interface after authentication
// General access test, login treated only as precondition; uses demo credentials

test('TC-US-001-01: Access leave workflow config after login', async ({ page }) => {
  await login(page);
  await openLeaveWorkflowConfig(page);
  // Validate config UI loaded
  await page.waitForSelector('text=Leave Types');
});

// TC-US-001-02: Create a new approval chain based on organizational template
// (Test plan mapped as adding new Leave Type in demo)
test('TC-US-001-02: Create new approval chain via template', async ({ page }) => {
  await login(page);
  await openLeaveWorkflowConfig(page);
  await page.getByRole('button', { name: ' Add' }).click();
  await page.locator('form').getByRole('textbox').fill('Demo Approval Chain');
  await page.getByRole('button', { name: 'Save' }).click();
  // Assertion based on observed UI, not hardcoded
  await page.waitForSelector('text=Successfully Saved'); // Actual message from UI
});
