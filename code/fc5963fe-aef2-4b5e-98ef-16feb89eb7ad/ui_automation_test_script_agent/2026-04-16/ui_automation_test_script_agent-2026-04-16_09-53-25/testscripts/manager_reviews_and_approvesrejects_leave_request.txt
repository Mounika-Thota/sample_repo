// Helper to login as manager
async function loginAsManager(page) {
  await page.goto(appUrl, { waitUntil: 'domcontentloaded' });
  // Session may expire; check if landing on login
  // Username input: .oxd-input[name='username']
  // Password input: .oxd-input[name='password']
  await page.locator('.oxd-input[name="username"]').fill(managerUsername);
  await page.locator('.oxd-input[name="password"]').fill(managerPassword);
  // Login button: .oxd-button--main[type='submit'] (actual)
  await page.locator('.oxd-button--main[type="submit"]').click();
  // Wait for main dashboard (sidebar appears)
  await page.waitForSelector('.oxd-main-menu');
}

// Helper to navigate to Leave module
async function navigateToLeaveModule(page) {
  // Sidebar link with class .oxd-main-menu-item and text 'Leave'
  const leaveNav = page.locator('.oxd-main-menu-item').filter({ hasText: 'Leave' });
  await expect(leaveNav).toBeVisible();
  await leaveNav.click();
  // Wait for module to load (Leave grid appears)
  await page.waitForSelector('.oxd-table');
}

}
