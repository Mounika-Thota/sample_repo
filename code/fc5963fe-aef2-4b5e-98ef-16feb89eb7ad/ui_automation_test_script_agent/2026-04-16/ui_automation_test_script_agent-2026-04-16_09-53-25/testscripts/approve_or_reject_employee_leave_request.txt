    // Navigate to Leave module via sidebar
    await page.locator('nav.oxd-sidepanel a.oxd-main-menu-item').filter({ hasText: 'Leave' }).click();
    await page.waitForURL('**/leave/viewLeaveList', { timeout: 15000 });
    // Click "Leave List" tab if present (OrangeHRM uses a tab with this text)
    if (await page.locator('.oxd-topbar-body-nav-tab').filter({ hasText: 'Leave List' }).count() > 0) {
      await page.locator('.oxd-topbar-body-nav-tab').filter({ hasText: 'Leave List' }).click();
    }
    // Ensure the "Pending" status filter is selected
    if (await page.locator('.oxd-checkbox-input + label').filter({ hasText: 'Pending Approval' }).count() > 0) {
      await page.locator('.oxd-checkbox-input + label').filter({ hasText: 'Pending Approval' }).click();
    }
    // Click "Search" to refresh the list
    if (await page.locator('button').filter({ hasText: 'Search' }).count() > 0) {
      await page.locator('button').filter({ hasText: 'Search' }).click();
    }
    // Wait for the results/data table
    await page.waitForSelector('.oxd-table', { timeout: 10000 });
    // If a pending request exists, click first one in the table
    if (await page.locator('.oxd-table-card').count() === 0) {
      test.skip('No pending leave requests exist; skipping test.');
    } else {
      // Click the first pending request for details
      const firstRequest = page.locator('.oxd-table-card').first();
      await firstRequest.click();
      // Wait for the detail overlay/modal (actual modal uses class .oxd-dialog-container)
      await page.waitForSelector('.oxd-dialog-container', { timeout: 10000 });
      // Adaptation: Use actual labels present in modal
      const fieldsToCheck = ['Employee Name', 'Leave Type', 'From Date', 'To Date'];
      for (const label of fieldsToCheck) {
        expect(await page.locator('.oxd-dialog-container').getByText(label).count()).toBeGreaterThan(0);
      }
    }
  });
